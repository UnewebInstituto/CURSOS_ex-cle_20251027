/* 
Objeto JavaScript.assign()
El Object.assign()método copia propiedades de uno o más objetos de origen a un objeto de destino.
*/

// Create Target Object
const person1 = {
  firstName: "John",
  lastName: "Doe",
  age: 50,
  eyeColor: "blue"
};

// Create Source Object
const person2 = {firstName: "Anne",lastName: "Smith"};

// Assign Source to Target
Object.assign(person1, person2);
console.log(person1);
